Program for THALAPAKATTI STYLE MUTTON BIRYANI RECIPE

South Indian one of most delious food is briyani. we going to make Thalapakatti briyani with special spices. with special ingredients and Mutton or Chicken...

import * as cooking './cooking_config';
import * as Vessel './cooking_Tools';
import * as Water './natural.resrc';

export soakedRice to_Get_Soak_Rice();

Let
rice = Jeeraga Samba rice 5 Cup;

Water.add.rice.wash;
Water.add.wait.rice.soak(TIMEOUT:15minits);

return(soakedRice);

export boiledMutton to_Get_PreCooked_Mutton();

Let
Meat = Mutton 1Kg

Water.add.Meat;
Water.add.boil.Meat.wait(TIMEOUT:5 whistle);

return(boiledMutton);

export masala get_Briyani_masala();

Let
Mace = 1 nos
Cashew = 15 nos;
Cinnamon = 4 nos
Star anise = 2 nos
Cloves = 4 nos
Cardamon = 5nos
Kalpasi = 3 leaf
Bay leaf = 2 leaf
Corriander whole seed = 2 tbsp
Cumin = 2 tbsp
Funnel Seed = 1 tbsp

fry.all_ingredients(TIMEOUT:10 Min);
switch off and let leave it for(TIMEOUT:10 Min);
put.all ingredients.MixerGrinder();;

return(masala);

export paste To_Grind_and_make_it_paste();

Let
Ginger = 10 grams
Garlic = 10 Cloves
Onion small = 200 grams
Green Chiili = 5nos

Wash.all_ingredients(TIMEOUT:10 Mintis);
put.all_ingredients.MixerGrinder();
return(paste);

export briyani Time for prepartion();

Let

oil = 150 grams
Ghee = 50 grams
Curd = 200 grams
Water = 7.5 Cup

turn.on.stove();
place.Vessel(TIMEOUT:2 Min);
pour.oil(TIMEOUT:10 Min);
to_Grind_and_make_it_paste().sauteing.until(color turn to brown);
to_Get_PreCooked_Mutton().add.salt_and_chilli_powder.sauteing(TIMEOUT:3 Min);
add.Curd.sauteing.wait(TIMEOUT:2 Min);
add.Water.boil.wait(TIMEOUT:5 Min);
add.to_Get_Sock_Rice()(TIMEOUT:15 Min);
verify_rice_semi_cooked();
put.lid.wait(TIMEOUT:30 Min);
add.Ghee(TIMEOUT:5 Mintis);

return(briyani);
